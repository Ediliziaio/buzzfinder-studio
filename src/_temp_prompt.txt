### Badge per Unibox (hook `useInboxCount`):

```tsx
// src/hooks/useInboxCount.ts
import { useEffect, useState } from "react";
import { supabase } from "@/lib/supabase";

export function useInboxCount() {
  const [unreadCount, setUnreadCount] = useState(0);

  useEffect(() => {
    const fetch = async () => {
      const { count } = await supabase
        .from("inbox_messages")
        .select("id", { count: "exact", head: true })
        .eq("letto", false)
        .eq("archiviato", false);
      setUnreadCount(count || 0);
    };

    fetch();

    // Realtime subscription
    const channel = supabase
      .channel("inbox-count")
      .on("postgres_changes", {
        event: "INSERT",
        schema: "public",
        table: "inbox_messages",
      }, () => fetch())
      .subscribe();

    return () => { supabase.removeChannel(channel); };
  }, []);

  return unreadCount;
}
```

### Componente helper `SettingField`:

```tsx
// src/components/settings/SettingField.tsx
interface SettingFieldProps {
  chiave: string;
  label: string;
  description?: string;
  type?: "text" | "password" | "number";
  placeholder?: string;
}

export function SettingField({ chiave, label, description, type = "text", placeholder }: SettingFieldProps) {
  const [value, setValue] = useState("");
  const [saving, setSaving] = useState(false);
  const [saved, setSaved] = useState(false);

  useEffect(() => {
    supabase
      .from("app_settings")
      .select("valore")
      .eq("chiave", chiave)
      .maybeSingle()
      .then(({ data }) => {
        if (data?.valore) setValue(data.valore);
      });
  }, [chiave]);

  const handleSave = async () => {
    setSaving(true);
    await supabase
      .from("app_settings")
      .upsert({ chiave, valore: value }, { onConflict: "chiave" });
    setSaving(false);
    setSaved(true);
    setTimeout(() => setSaved(false), 2000);
  };

  return (
    <div className="space-y-1">
      <label className="text-sm font-medium">{label}</label>
      {description && <p className="text-xs text-muted-foreground">{description}</p>}
      <div className="flex gap-2">
        <Input
          type={type}
          value={value}
          onChange={(e) => setValue(e.target.value)}
          placeholder={placeholder}
          className="font-mono text-sm"
        />
        <Button
          size="sm"
          variant={saved ? "ghost" : "outline"}
          onClick={handleSave}
          disabled={saving}
        >
          {saving ? <Loader2 className="h-4 w-4 animate-spin" /> :
           saved ? <Check className="h-4 w-4 text-green-600" /> :
           "Salva"}
        </Button>
      </div>
    </div>
  );
}
```

### Componente helper `SettingToggle`:

```tsx
// src/components/settings/SettingToggle.tsx
interface SettingToggleProps {
  chiave: string;
  label: string;
  description?: string;
  defaultValue?: boolean;
}

export function SettingToggle({ chiave, label, description, defaultValue = false }: SettingToggleProps) {
  const [enabled, setEnabled] = useState(defaultValue);

  useEffect(() => {
    supabase
      .from("app_settings")
      .select("valore")
      .eq("chiave", chiave)
      .maybeSingle()
      .then(({ data }) => {
        if (data) setEnabled(data.valore === "true");
      });
  }, [chiave]);

  const toggle = async (checked: boolean) => {
    setEnabled(checked);
    await supabase
      .from("app_settings")
      .upsert({ chiave, valore: String(checked) }, { onConflict: "chiave" });
  };

  return (
    <div className="flex items-center justify-between">
      <div>
        <p className="text-sm font-medium">{label}</p>
        {description && <p className="text-xs text-muted-foreground">{description}</p>}
      </div>
      <Switch checked={enabled} onCheckedChange={toggle} />
    </div>
  );
}
```

## PARTE 5 — Hook globale `useAppSettings`

Crea un hook React che carica tutti i settings dal DB una volta sola e li mette in cache:

```tsx
// src/hooks/useAppSettings.ts
import { useState, useEffect, createContext, useContext } from "react";
import { supabase } from "@/lib/supabase";
import type { AppSetting } from "@/types";

type SettingsMap = Record<string, string>;

const AppSettingsContext = createContext<{
  settings: SettingsMap;
  getSetting: (chiave: string) => string | undefined;
  isLoaded: boolean;
}>({ settings: {}, getSetting: () => undefined, isLoaded: false });

export function AppSettingsProvider({ children }: { children: React.ReactNode }) {
  const [settings, setSettings] = useState<SettingsMap>({});
  const [isLoaded, setIsLoaded] = useState(false);

  useEffect(() => {
    supabase
      .from("app_settings")
      .select("chiave, valore")
      .then(({ data }) => {
        const map: SettingsMap = {};
        data?.forEach((s: AppSetting) => { map[s.chiave] = s.valore; });
        setSettings(map);
        setIsLoaded(true);
      });
  }, []);

  return (
    <AppSettingsContext.Provider value={{
      settings,
      getSetting: (chiave) => settings[chiave],
      isLoaded,
    }}>
      {children}
    </AppSettingsContext.Provider>
  );
}

export const useAppSettings = () => useContext(AppSettingsContext);
```

Avvolgi l'app con `<AppSettingsProvider>` in `main.tsx` o `App.tsx`.

## NOTE IMPORTANTI

### Schema DB — Nomi colonne corretti:
- `app_settings` usa `chiave` e `valore` (NON key/value)
- `campaign_recipients` usa `contact_id` (NON contatto_id)
- `contacts` ha `sito_web` (NON sito)
- Tutti i campi booleani usano il nome italiano: `solo_lavorativi`, `tracking_aperture`, ecc.

### Supabase Realtime:
- Per l'Unibox, abilita Realtime su `inbox_messages` nella Supabase Dashboard
- Per le statistiche live, abilita Realtime su `campaign_step_executions`

### RLS (Row Level Security):
- Tutti gli INSERT/SELECT nelle pagine frontend usano il client `supabase` con la anon key
- Le Edge Functions usano `SUPABASE_SERVICE_ROLE_KEY` per bypassare RLS
- Assicurati che le policy RLS permettano l'accesso authenticated agli utenti

### Edge Functions URL:
- `process-sequence`: POST `/functions/v1/process-sequence`
- `track-event`: GET `/functions/v1/track-event` (PUBLIC — no JWT)
- `handle-reply`: POST `/functions/v1/handle-reply` (PUBLIC — no JWT)
- `validate-emails`: POST `/functions/v1/validate-emails`
- `assign-senders`: POST `/functions/v1/assign-senders`
